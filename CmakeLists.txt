cmake_minimum_required(VERSION 3.10)
project(GlobalTimeZoneApp VERSION 1.0.0 LANGUAGES CXX)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find CURL package
find_package(CURL REQUIRED)
include_directories(${CURL_INCLUDE_DIRS})

# Add nlohmann/json as a header-only library
include_directories(${CMAKE_SOURCE_DIR}/external/json/include)

# Specify source files
set(SOURCES
    main.cpp
    TimeZoneApiClient.cpp
    TimeZoneApp.cpp
)

# Specify header files
set(HEADERS
    TimeZone.h
    TimeZoneApiClient.h
    TimeZoneApp.h
)

# Create executable
add_executable(TimeZoneApp ${SOURCES} ${HEADERS})

# Link libraries
target_link_libraries(TimeZoneApp PRIVATE ${CURL_LIBRARIES})

# If C++17 filesystem is not integrated in the standard library on your platform
if(CMAKE_CXX_COMPILER_ID STREQUAL "GNU" AND CMAKE_CXX_COMPILER_VERSION VERSION_LESS 9.0)
    target_link_libraries(TimeZoneApp PRIVATE stdc++fs)
endif()

# Create data directory in build directory
add_custom_command(
    TARGET TimeZoneApp POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E make_directory "${CMAKE_BINARY_DIR}/data"
)

# Install targets
install(TARGETS TimeZoneApp DESTINATION bin)
